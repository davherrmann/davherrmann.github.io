canvas(({context, width, height}) => {
  context.beginPath();
  context.arc(
    width / 2,
    height / 2,
    Math.min(width, height) * 0.3,
    0,
    Math.PI * 2,
    true
  )
  context.fillStyle = '#49045f';
  context.fill();
})

  const material = new THREE.MeshNormalMaterial()

  const mesh = new THREE.Mesh(geometry, material)
  scene.add(mesh)

  render(() => {
    mesh.rotation.x += 0.01;
